import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("==COMPARADOR DE EFICIENCIA: MERGE SORT VS BUBBLE SORT==");
        System.out.print("Ingrese la cantidad de elementos del vector: ");
        int tamano = scanner.nextInt();

        int[] vectorOriginal = new int[tamano];

        System.out.println("\n--- Ingrese los elementos ---");
        for (int i = 0; i < tamano; i++) {
            System.out.print("Elemento [" + i + "]: ");
            vectorOriginal[i] = scanner.nextInt();
        }

        int[] vectorParaBubble = vectorOriginal.clone();
        int[] vectorParaMerge = vectorOriginal.clone();

        System.out.println("\nVector original ingresado:");
        imprimirVector(vectorOriginal);

        System.out.println("\n--- Ejecutando Ordenamientos ---");

        long inicioBubble = System.nanoTime();
        bubbleSort(vectorParaBubble);
        long finBubble = System.nanoTime();
        long tiempoBubble = finBubble - inicioBubble;

        long inicioMerge = System.nanoTime();
        mergeSort(vectorParaMerge, 0, vectorParaMerge.length - 1);
        long finMerge = System.nanoTime();
        long tiempoMerge = finMerge - inicioMerge;

        System.out.println("\nVector ordenado final:");
        imprimirVector(vectorParaMerge); 

        System.out.println("\n=== TABLA DE EFICIENCIA ===");
        System.out.println("Algoritmo\tTiempo de ejecución (Nanosegundos)");
        System.out.println("Bubble Sort\t" + tiempoBubble + " ns");
        System.out.println("Merge Sort\t" + tiempoMerge + " ns");
        scanner.close();
    }

    public static void mergeSort(int[] arr, int izquierda, int derecha) {
        if (izquierda < derecha) {
            int medio = izquierda + (derecha - izquierda) / 2;

            mergeSort(arr, izquierda, medio);
            mergeSort(arr, medio + 1, derecha);

            mezclar(arr, izquierda, medio, derecha);
        }
    }

    public static void mezclar(int[] arr, int izquierda, int medio, int derecha) {
        int n1 = medio - izquierda + 1;
        int n2 = derecha - medio;

        int[] izqArr = new int[n1];
        int[] derArr = new int[n2];

        for (int i = 0; i < n1; ++i) izqArr[i] = arr[izquierda + i];
        for (int j = 0; j < n2; ++j) derArr[j] = arr[medio + 1 + j];

        int i = 0, j = 0;
        int k = izquierda;

        while (i < n1 && j < n2) {
            if (izqArr[i] <= derArr[j]) {
                arr[k] = izqArr[i];
                i++;
            } else {
                arr[k] = derArr[j];
                j++;
            }
            k++;
        }

        while (i < n1) {
            arr[k] = izqArr[i];
            i++;
            k++;
        }
        while (j < n2) {
            arr[k] = derArr[j];
            j++;
            k++;
        }
    }

    public static void bubbleSort(int[] vector) {
        int n = vector.length;
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (vector[j] > vector[j + 1]) {
                    int aux = vector[j];
                    vector[j] = vector[j + 1];
                    vector[j + 1] = aux;
                }
            }
        }
    }

    public static void imprimirVector(int[] arr) {
        for (int num : arr) System.out.print(num + " ");
        System.out.println();
    }
}
